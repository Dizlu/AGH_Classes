#include <iostream>
using namespace std;
#include <time.h>

/******************************************************/
float *fun(int **tab, int row, int col){
	// Funkcja otrzymujaca wskaünik do tablicy 2D, jej dlguosci i szerokosci
	// Tworzymy wektor 1D dla zwrocenia wyniku
	float *result = new float[col];
	// Petla sumujaca liczby znajdujace sie w danym rzedzie
	for (int i = 0; i < row; i++){
		float wynik = 0;

		for (int j = 0; j < col; j++){
			wynik += tab[i][j];
		}
		result[i] = wynik / col;
	}
	return result;
}
/******************************************************/

int main(){
	srand( time(NULL));
	// Tworzenie zmiennych do wpisania przez uzytkownika
	int n, m;
	cin >> n;
	cin >> m;
	// Utowrzenie macierzy 2D
	int **w = new int*[n];
	for(int i = 0; i < n; i++){
		w[i] = new int[m];
	}
	// Wypelnienie macierzy randomami
	for (int i = 0; i < n; i++){
		for (int j = 0; j < m; j++){
			w[i][j] = rand() % 10;
		}
	}
	// Wydrukowanie pierwszej macierzy wypelnionej randomami
	cout << " Pierwsza macierz to: " << endl;
	for (int i = 0; i < n; i++){
		cout << "| ";
		for (int j = 0; j < m; j++){
			cout << w[i][j] << " ";
		}
		cout << " | \n";
	}
	// Wywolanie kolejnej funkcji
	cout << "Sumy kolejnych rzedow danej macierzy to: \n " ;
	float *a = fun(w, n, m);
	for (int i = 0; i < n; i++){
		cout << a[i] << "\n";
	}

}
